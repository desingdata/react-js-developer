01

Renderización de elementos

React convierte elementos JSX en nodos del DOM real. Cada vez que el estado o las props cambian, React compara el Virtual DOM con el DOM actual y actualiza solo lo necesario (reconciliación).

App.jsx 

// RENDERIZACIÓN DE ELEMENTOS
// React transforma JSX en llamadas a React.createElement().
// El resultado es un objeto (Virtual DOM) que React compara
// con el DOM real para aplicar solo los cambios necesarios.

import React from 'react';
import { createRoot } from 'react-dom/client';

// Elemento JSX simple
const elemento = (
  <ul className="lista">
    <li>⚡ Renderizado inicial</li>
    <li>🔁 Re-render por cambio de estado</li>
    <li>🎯 Solo actualiza lo que cambió</li>
  </ul>
);

// Renderización: monta el árbol en el DOM
const root = createRoot(
  document.getElementById('root')
);
root.render(elemento);

// React NO vuelve a crear los nodos existentes,
// solo parchea las diferencias (diffing algorithm).

02

Componentes y propiedades

Un componente es una función que recibe un objeto props y retorna JSX. Las props son de solo lectura: los datos fluyen hacia abajo (one-way data binding).

UsuarioCard.jsx 

// COMPONENTES Y PROPIEDADES (PROPS)
// Un componente funcional es una función de JavaScript
// que recibe "props" y retorna JSX.
// Las props son de SOLO LECTURA: no se deben mutar.

// Componente reutilizable: recibe datos por props
function UsuarioCard({ nombre, rol, emoji, color }) {
  return (
    // El estilo inline usa la prop "color"
    <div style={{ borderColor: color }}>
      <span className="avatar">{emoji}</span>
      <div>
        <strong>{nombre}</strong>   {/* prop nombre */}
        <span>{rol}</span>          {/* prop rol    */}
      </div>
    </div>
  );
}

// Uso: se pasan props como atributos HTML
function App() {
  return (
    <>
      <UsuarioCard
        nombre="Ana García"
        rol="Frontend Dev"
        emoji="👩‍💻"
        color="#5eead4"
      />
      <UsuarioCard
        nombre="Luis Torres"
        rol="Backend Dev"
        emoji="🧑‍💻"
        color="#fb923c"
      />
    </>
  );
}

Estado y ciclo de vida

useState añade memoria interna al componente. useEffect sincroniza efectos secundarios y representa el ciclo de vida: montar, actualizar y desmontar.

Contador.jsx 

// ESTADO Y CICLO DE VIDA
// useState: añade una variable reactiva al componente.
//   - [valor, setter] = useState(valorInicial)
//   - Al llamar al setter React programa un re-render.
//
// useEffect: corre código como efecto secundario.
//   - [] (vacío) → solo al montar  (componentDidMount)
//   - [dep]      → al montar + cuando dep cambia (didUpdate)
//   - return fn  → al desmontar   (componentWillUnmount)

import { useState, useEffect } from 'react';

function Contador() {
  // Estado: "count" empieza en 0
  const [count, setCount] = useState(0);
  const [log,   setLog]   = useState([]);

  // Ciclo de vida: se ejecuta al montar
  useEffect(() => {
    agregarLog('🟢 Componente montado (mount)');

    // Cleanup: se ejecuta al desmontar
    return () => agregarLog('🔴 Desmontado (unmount)');
  }, []); // [] = solo al montar

  // Se ejecuta cada vez que "count" cambia
  useEffect(() => {
    if (count > 0)
      agregarLog(`🔄 count actualizado → ${count}`);
  }, [count]); // depende de count

  const agregarLog = (msg) =>
    setLog(prev => [msg, ...prev].slice(0, 6));

  return (
    <div>
      <h2>{count}</h2>
      <button onClick={() => setCount(c => c + 1)}>+1</button>
      <button onClick={() => setCount(c => c - 1)}>−1</button>
      <button onClick={() => setCount(0)}>Reset</button>
      <ul>{log.map((l,i) => <li key={i}>{l}</li>)}</ul>
    </div>
  );
}

04

Manejo de eventos

React usa eventos sintéticos (SyntheticEvent) que envuelven los eventos nativos del navegador para garantizar compatibilidad cruzada. Los manejadores se pasan como props en camelCase.

Formulario.jsx 

// MANEJO DE EVENTOS
// React usa SyntheticEvent: una capa de abstracción sobre
// los eventos nativos del navegador para garantizar
// comportamiento consistente en todos los navegadores.
//
// Reglas clave:
//  • Los manejadores van en camelCase: onClick, onChange
//  • Se pasan como funciones, NO como strings
//  • e.preventDefault() previene el comportamiento nativo
//  • e.stopPropagation() detiene la propagación (bubbling)

import { useState } from 'react';

function Formulario() {
  const [texto, setTexto] = useState('');

  // onChange: se dispara en cada pulsación de tecla
  const handleChange = (e) => {
    setTexto(e.target.value); // e.target.value = valor actual
  };

  // onClick: captura el clic en el botón
  const handleClick = (e) => {
    // e.currentTarget = el elemento que tiene el listener
    console.log('Clic en:', e.currentTarget.textContent);
  };

  // onSubmit: previene el reload por defecto del form
  const handleSubmit = (e) => {
    e.preventDefault(); // evita recarga de página
    alert(`Enviado: "${texto}"`);
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        value={texto}
        onChange={handleChange}  // evento onChange
        placeholder="Escribe algo..."
      />
      <p>Vista previa: {texto}</p>
      <button
        type="button"
        onClick={handleClick}    // evento onClick
      >
        Clic aquí
      </button>
    </form>
  );
}
